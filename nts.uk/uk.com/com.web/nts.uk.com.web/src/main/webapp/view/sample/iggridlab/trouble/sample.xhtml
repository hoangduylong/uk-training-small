No 1 When create grid in dialog with javascript code, If the javascript code almost does nothing, process is very fast, before the dialog is opened, the grid has a some calculation logics that is incorrect. Use _.defer in __viewContext.ready. Sample (From KAL001): 
__viewContext.ready(function() {
		        let param =  nts.uk.ui.windows.getShared("extractedAlarmData");
		        let screenModel = new viewmodel.ScreenModel(param);
		
		        _.defer(function (){
		            screenModel.startPage().done(function() {
		                __viewContext.bind(screenModel);            
		                $("#grid").igGrid("option", "dataSource", extractedAlarmData);
		//            screenModel.dataSource= extractedAlarmData;
		            });
		        });
		    });
