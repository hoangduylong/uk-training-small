Internationalization

Use in java

					//scope annotation
					//param can be plain text or resource id
					public class A{
						public void greeting(){
							System.out.println(TextResources.localize("code"));
							System.out.println(TextResources.localize("code","param1","param2",......));
							System.out.println(TextResources.localize("code","#resource_id1","#resource_id2",......));
							System.out.println(TextResources.localize("code",new String[]{"a","b"}));
						}
					}
